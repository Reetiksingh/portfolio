// Role-based access middleware
const authorize = (role) => {
  return (req, res, next) => {
    if (req.user.role !== role)
      return res.status(403)
        .json({ error: 'Access denied' });
    next();
  };
};

// Barcode lookup route
router.get('/nutrition/:code',
  authenticate,
  authorize('user'),
  NutritionController.scan
);
